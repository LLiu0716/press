# 游戏主页面

<template>
  <main class="main">
    <div>
      <div class="btn-box">
        <button
          v-for="v in button"
          :key="v.id"
          :class="{ checked: checked == v.id }"
          @click="onChecked(v.id)"
        >
          {{ v.value }}
        </button>
        <button @click="onReset">重置</button>
      </div>
      <Mine
        :td="num"
        :tr="num"
        :mine-num="mineNum"
        :reset="reset"
        @onReset="onReset"
        @onNext="onNext"
      />
    </div>
  </main>
</template>

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import Mine from './MineClearance/Mine.vue'

const checked = ref(1)
const reset = ref(false)

const num = computed(() => {
  return checked.value * 7
})

const mineNum = computed(() => {
  switch (checked.value) {
    case 1:
      return 9
    case 2:
      return 40
    default:
      return 99
  }
})

// const button = ref<Array<Btn>>([
const button = ref([
  {
    id: 1,
    value: '简单'
  },
  {
    id: 2,
    value: '中等'
  },
  {
    id: 3,
    value: '困难'
  }
])

const onChecked = (id: number) => {
  checked.value = id
  reset.value = !reset.value
}

const onReset = () => {
  reset.value = !reset.value
}

const onNext = () => {
  const index = button.value.findIndex(v => v.id === checked.value)
  if (index + 1 != button.value.length) {
    checked.value = button.value[index + 1].id
  } else {
    checked.value = button.value[0].id
  }
  onReset()
}

onMounted(() => {})
</script>

# 雷页面

<template>
  <div class="box" @contextmenu.prevent="contextmenu">
    <table class="mine-box">
      <tbody :class="{ tbody: mineNum > 30 }">
        <tr v-for="(item, index) in content" :key="index">
          <td
            v-for="(v, i) in item"
            :key="`${index}-${i}`"
            :class="[v.type === 'number' ? `number-${v.value}` : v.type]"
            @mousedown="checkedTd($event, v, index, i)"
          >
            {{ v.value ? v.value : '' }}
          </td>
        </tr>
      </tbody>
    </table>
    <p>
      <span>剩余数量</span>&nbsp;<b :class="`number-${surplusMineNum}`">
        {{ surplusMineNum }}
      </b>
    </p>
    <div v-show="over" class="over" />
    <div v-show="!surplusMineNum" class="alert-box">
      <div class="alert">
        <h1 class="content">
          <strong>{{ surplus }}</strong>
        </h1>
        <div :class="['btn', { next: surplus != '游戏失败' }]">
          <button @click="emit('onReset')">重新开始</button>
          <button v-if="surplus != '游戏失败'" @click="emit('onNext')">
            增加难度
          </button>
        </div>
      </div>
    </div>
  </div>
</template>

<script lang="ts" setup>
import { onMounted, ref, watch } from 'vue'
import { mine } from './mine'

type Table = {
  id: number
  type: string
  value: number
  checked: boolean
}

const props = withDefaults(
  defineProps<{
    tr: number
    td: number
    mineNum: number
    reset: boolean
  }>(),
  {
    tr: 9,
    td: 9,
    mineNum: 10
  }
)

const contextmenu = () => {
  return false
}

watch(
  () => props.reset,
  () => {
    init()
  }
)

const content = ref<[Table][]>([])
const over = ref(false)
const surplus = ref('恭喜过关')
const surplusMineNum = ref(0)

const emit = defineEmits(['onReset', 'onNext'])

const { data, getSquares, getAround } = mine()

const init = () => {
  over.value = false
  content.value = []
  surplusMineNum.value = props.mineNum
  getSquares(props.td, props.tr, props.mineNum)
  let n = 1
  content.value = data.squares.map((item) => {
    return item.map(() => {
      return {
        id: n++,
        type: '',
        value: 0,
        checked: false
      }
    }) as [Table]
  }) as []
}

const checkedTd = (e: MouseEvent, v: Table, y: number, x: number) => {
  if (e.buttons == 2) {
    mouseRight(v, y, x)
  } else {
    mouseLeft(v, y, x)
  }
}

const mouseRight = (v: Table, y: number, x: number) => {
  if (!surplusMineNum.value) return
  const square = data.squares[y][x]
  if (v.type && v.type !== 'flag') {
    return
  }
  if (v.type == 'flag') {
    v.type = ''
    surplusMineNum.value++
    data.checkedList[surplusMineNum.value] = ''
  } else {
    v.type = 'flag'
    surplusMineNum.value--
    data.checkedList[surplusMineNum.value] = square.type
  }
  if (!surplusMineNum.value) {
    const result = data.checkedList.some((v) => v == 'number')
    over.value = true
    if (result) {
      surplus.value = '游戏失败'
    } else {
      surplus.value = '恭喜过关'
    }
  }
}

const mouseLeft = (v: Table, y: number, x: number) => {
  if (v.type) return
  const square = data.squares[y][x]
  v.type = square.type
  v.value = square.value
  v.checked = true
  if (square.type == 'mine') {
    over.value = true
    for (let i = 0; i < props.tr; i++) {
      for (let j = 0; j < props.td; j++) {
        if (data.squares[i][j].type === 'mine') {
          content.value[i][j].type = 'mine'
        }
      }
    }
    v.type += ' game-over'
    surplusMineNum.value = 0
    surplus.value = '游戏失败'
    return
  }
  // 递归出周围的格子的数值
  const getAllZero = (squareZero: { x: number; y: number }) => {
    const around = getAround(squareZero, props.td, props.tr)
    around.forEach((item) => {
      const x = item[0]
      const y = item[1]
      content.value[x][y].type = 'number'
      content.value[x][y].value = data.squares[x][y].value
      if (data.squares[x][y].value == 0) {
        if (!data.squares[x][y].check) {
          data.squares[x][y].check = true
          getAllZero(data.squares[x][y])
        }
      }
    })
  }
  getAllZero(square)
}

onMounted(() => {
  init()
})
</script>

# 排雷逻辑

type Table = {
  check?: boolean
  type: string
  value: number
  x: number
  y: number
}

type IData = {
  squares: [Table][]
  checkedList: string[]
  surplusMine: number
}

type ISquare = {
  x: number
  y: number
  [key: string]: unknown
}

export const mine = () => {
  const data: IData = {
    squares: [],
    checkedList: [],
    surplusMine: 0
  }

  // 生成 N 个不重复的数字
  const randomNum = (tr: number, td: number, mineNum: number) => {
    return Array.from({ length: tr * td }, (v, i) => i)
      .sort(() => 0.5 - Math.random())
      .slice(0, mineNum)
  }

  // 找周围的格子
  const getAround = (square: ISquare, td: number, tr: number) => {
    /*
      位置逻辑 
      x-1,y-1    x,y-1    x+1,y-1
       x-1,y      x,y      x+1,y
      x-1,y+1    x,y+1    x+1,y+1
    */
    const { x, y } = square
    const result = []
    for (let i = x - 1; i <= x + 1; i++) {
      for (let j = y - 1; j <= y + 1; j++) {
        if (
          i < 0 || // 超出了左边的范围
          j < 0 || // 超出了上边的范围
          i > td - 1 || // 超出了右边的范围
          j > tr - 1 || // 超出了下边的范围
          (i == x && j == y) || // 自己
          data.squares[j][i].type === 'mine' // 雷
        ) {
          continue
        }
        result.push([j, i])
      }
    }
    return result
  }

  // 更新周围格子的参数
  const updateNum = (tr: number, td: number) => {
    for (let i = 0; i < tr; i++) {
      for (let j = 0; j < td; j++) {
        // 只更新雷周围的数字
        if (data.squares[i][j].type === 'mine') {
          const num = getAround(data.squares[i][j], td, tr)
          for (let k = 0; k < num.length; k++) {
            data.squares[num[k][0]][num[k][1]].value += 1
          }
        }
      }
    }
  }

  const getSquares = (tr: number, td: number, mineNum: number) => {
    init()
    data.checkedList = Array.from({ length: mineNum }, () => '')
    data.surplusMine = mineNum
    const mineNumList = randomNum(tr, td, mineNum)
    let n = 0
    for (let i = 0; i < tr; i++) {
      data.squares[i] = [] as unknown[] as [Table]
      for (let j = 0; j < td; j++) {
        const obj = {
          x: j,
          y: i,
          type: 'number',
          value: 0
        }
        if (mineNumList.includes(n++)) {
          obj.type = 'mine'
        }
        data.squares[i][j] = obj
      }
    }
    updateNum(tr, td)
  }

  const init = () => {
    data.checkedList = []
    data.squares = []
    data.surplusMine = 0
  }

  return {
    data,
    getAround,
    getSquares
  }
}
