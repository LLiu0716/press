# 使用 vue3 实现经典游戏之贪吃蛇 包含代码

# 经典游戏之贪吃蛇

当前长度 0

提示: 长度超过 300 自动加速

# 完整代码

# 游戏主页面

<script setup lang="ts">
import { ref, onMounted, watch, onBeforeUnmount } from 'vue'
import Snake from './GluttonousSnake/Snake.vue'
import Control from './GluttonousSnake/Control.vue'
import Alert from './Alert.vue'

const size = 18 // 大小
const index = 1 // 倍数
const total = 300 // 通关长度

const config = {
  /**
   * 方向信息
   * 1. ↑
   * 2: →
   * 3: ←
   * 4: ↓
   */
  direction: '2',
  max: size * index * 20
}

/** 蛇的相关数据 */
const snakeList = ref<
  {
    type: 'head' | 'body'
    id: number
    control: Record<string, string>
  }[]
>([])

/** 游戏是否结束 */
const hansGameOver = ref(false)

/** 游戏 开始/暂停 */
const suspend = ref(false)

/** 食物的位置信息 */
const target = ref({
  top: '',
  left: ''
})

/** 如果蛇的长度大于通关长度就开始下一关 */
watch(
  () => snakeList.value.length,
  value => {
    if (value > total) {
      select(false)
    }
  }
)

/** 点击方向按钮 */
const onClick = (direction: string) => {
  if (direction === 'Space') {
    start()
    return
  }
  if (!suspend.value) return
  if (config.direction === '1' && direction === '4') return
  if (config.direction === '2' && direction === '3') return
  if (config.direction === '3' && direction === '2') return
  if (config.direction === '4' && direction === '1') return
  config.direction = direction
  move()
}

let timeId: number

/** 速度 */
let speed = 360

const start = () => {
  suspend.value = !suspend.value
  if (!suspend.value) {
    document.body.style.overflow = 'auto'
    clearInterval(timeId)
  } else {
    document.body.style.overflow = 'hidden'
    timeId = setInterval(move, speed)
  }
}

/** 初始化 */
const info = () => {
  config.direction = '2'
  snakeList.value = []
  for (let i = 0; i < 5; i++) {
    snakeList.value.push({
      id: 0 + i,
      control: {
        top: '0px',
        left: size * i + 'px',
        [i === 4 ? 'transform' : '']: 'rotate(-90deg)'
      },
      type: i === 4 ? 'head' : 'body'
    })
  }
  randomTarget()
  clearInterval(timeId)
  if (!suspend.value) return
  suspend.value = true
  timeId = setInterval(move, speed)
}

/** 加速或者减速 */
const select = (up: boolean) => {
  if (speed > 80 && up) {
    speed -= 60
  }
  if (speed < 800 && !up) {
    speed += 60
  }
  info()
}

/** 随机生成食物位置信息 */
const randomNumber = (): any => {
  const i = ~~(Math.random() * 19) + 1
  const j = ~~(Math.random() * 19) + 1
  if (
    !snakeList.value.some(
      v =>
        parseInt(v.control.top!) === i * size &&
        parseInt(v.control.left!) === j * size
    )
  ) {
    return { top: i * size + 'px', left: size * j + 'px' }
  } else {
    return randomNumber()
  }
}

/** 生成食物 */
const randomTarget = () => {
  const { top, left } = randomNumber()
  target.value = {
    top,
    left
  }
}

/** 移动蛇 */
const move = () => {
  if (
    snakeList.value.some(
      v =>
        v.control.top === target.value.top &&
        v.control.left === target.value.left
    )
  ) {
    snakeList.value.push({
      type: 'body',
      id: snakeList.value.length,
      control: target.value
    })
    randomTarget()
  }
  const item = snakeList.value.find(v => v.type === 'head')
  const obj = {
    top: '',
    left: '',
    transform: ''
  }
  switch (config.direction) {
    case '1':
      obj.transform = 'rotate(-180deg)'
      obj.top = parseInt(item?.control.top!) - size + 'px'
      obj.left = item?.control.left!
      break
    case '2':
      obj.transform = 'rotate(-90deg)'
      obj.left = parseInt(item?.control.left!) + size + 'px'
      obj.top = item?.control.top!
      break
    case '3':
      obj.transform = 'rotate(90deg)'
      obj.left = parseInt(item?.control.left!) - size + 'px'
      obj.top = item?.control.top!
      break
    default:
      obj.transform = 'rotate(0deg)'
      obj.top = parseInt(item?.control.top!) + size + 'px'
      obj.left = item?.control.left!
      break
  }
  if (
    parseInt(obj.top) < 0 ||
    parseInt(obj.top) >= config.max ||
    parseInt(obj.left) < 0 ||
    parseInt(obj.left) >= config.max ||
    snakeList.value.some(
      v => v.control.top === obj.top && v.control.left === obj.left
    )
  ) {
    hansGameOver.value = true
    isShow.value = true
    return
  }
  item!.type = 'body'
  item!.control.transform = ''
  snakeList.value.push({
    type: 'head',
    id: snakeList.value[0].id,
    control: obj
  })
  snakeList.value.splice(0, 1)
}
const isShow = ref(false)

function onConfirm() {
  if (hansGameOver.value) {
    suspend.value = true
    isShow.value = false
    info()
  } else {
    select(false)
  }
}

onMounted(() => {
  info()
})

onBeforeUnmount(() => {
  document.body.style.overflow = ''
})
</script>

<template>
  <main class="main">
    <div
      class="main-view"
      :style="{ minWidth: 365 * index + 'px', minHeight: 365 * index + 'px' }"
    >
      <Snake
        v-for="(v, i) in snakeList"
        :key="v.id"
        :index="i"
        :type="v.type"
        :width="size * index"
        :height="size * index"
        :control="v.control"
      />
      <div
        class="target"
        :style="{
          width: size * index + 'px',
          height: size * index + 'px',
          ...target
        }"
      />
    </div>
    <div class="main-count">
      <button @click="select(false)">上一关</button>
      <span
        >当前长度
        <b :class="`snakeList-length-${snakeList.length % 3}`">
          {{ snakeList.length }}
        </b>
      </span>
      <button @click="select(true)">下一关</button>
    </div>
    <div v-if="index === 1" class="main-control">
      <button @click="start">{{ suspend ? '暂停' : '开始' }}</button>
      <Control @on-click="onClick" />
      <button @click="info">重来</button>
    </div>
    <p class="main-tip">
      <b>提示:</b> 长度超过
      <span>{{ total }}</span>
      自动加速
    </p>
  </main>
  <Alert v-model:value="isShow" @close="onConfirm">
    <p class="alert"><b>游戏失败</b></p>
  </Alert>
</template>

# 控制页面

<script setup lang="ts">
import { ref, onMounted, onBeforeUnmount } from 'vue'
import base from '../base'

const emits = defineEmits(['onClick'])
const touchstart = ref('')

const { isH5 } = base()

const addEventListener = () => {
  window.addEventListener('keydown', e => {
    let key = '2'
    switch (e.code) {
      case 'Space':
        key = 'Space'
        break
      case 'ArrowUp':
      case 'KeyW':
        key = '1'
        break
      case 'ArrowDown':
      case 'KeyS':
        key = '4'
        break
      case 'ArrowLeft':
      case 'KeyA':
        key = '3'
        break
      default:
        key = '2'
        break
    }
    emits('onClick', key)
  })
}

const onClick = (target: HTMLElement) => {
  touchstart.value = target.innerText
  emits('onClick', target.innerText)
}

onMounted(() => {
  if (isH5.value) return
  addEventListener()
})

onBeforeUnmount(() => {
  window.removeEventListener('keydown', () => {})
})
</script>

<template>
  <div
    v-if="isH5"
    class="control"
    @touchstart="e => onClick(e.target as HTMLElement)"
    @touchend="() => (touchstart = '')"
  >
    <span v-for="v in 4" :class="[touchstart === `${v}` ? 'check' : '']">
      {{ v }}
    </span>
  </div>
</template>

# 页面监听逻辑

import { ref, watch, onMounted } from 'vue'

const bast = () => {
  const resize = ref(0)
  const isH5 = ref(false)
  let timeId = true

  onMounted(() => {
    // 窗口发生改变才能触发监听 , 所以先让监听的值发生改变
    resize.value = window.innerWidth
    window.onresize = () => {
      return (() => {
        if (timeId) {
          timeId = false
          return window.setTimeout(() => {
            resize.value = window.innerWidth
            timeId = true
          }, 300)
        }
        return null
      })()
    }
  })

  // 这样可以关闭监听
  const outWatch = watch(resize, value => {
    const isAndroid = window.navigator.userAgent.match(/android/gi)
    const isIPhone = window.navigator.userAgent.match(/iphone/gi)
    if (isIPhone || isAndroid || value <= 750) {
      isH5.value = true
    } else {
      isH5.value = false
    }
  })

  return {
    // 监听窗口大小
    isH5,
    // 关闭监听
    outWatch
  }
}

export default bast
